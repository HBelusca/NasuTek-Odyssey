
set_rc_compiler()
spec2def(serialui.dll serialui.spec)

    serialui.c
    serialui.rc
    ${CMAKE_CURRENT_BINARY_DIR}/serialui.def)

add_cd_file(TARGET serialui DESTINATION odyssey/system32 FOR all)
